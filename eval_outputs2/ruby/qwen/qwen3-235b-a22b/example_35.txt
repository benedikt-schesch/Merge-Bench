      # Sup distance, or L-intinity norm
      # Parameters a and b are vectors with continuous attributes.      
      def sup_distance(a, b)
        distance = 0.0
        a.each_with_index do |item_a, i|
          item_b = b[i]
          diff = (item_a - item_b).abs
          distance = diff if diff > distance
        end
        distance
      end
